> FFmpeg多媒体文件处理

## 开发环境搭建

在这里我选择了Qt作为开发工具，因为后期可能会需要制作可视化界面，利用Qt会比较方便。Qt的安装步骤在这里就不会细讲，网上有大量的安装教程。在成功安装Qt后，我们需要使用FFmpeg。。

从[https://ffmpeg.zeranoe.com/builds/ ](https://ffmpeg.zeranoe.com/builds/ )下载FFmpeg安装包，static是静态库版本，shared是动态库版本，dev则是提供给开发者使用的lib文件。

在Qt中如何引入FFmpeg：首先是将解压ffmpeg中的文件只留下include和lib这两个文件夹，直接复制到我们的工程目录下；然后在Qt Creator中修改配置文件`.pro`，在其中加入：

```C++
INCLUDEPATH += $$PWD/ffmpeg/include

# 依赖模块
LIBS += \
    -L$$PWD/ffmpeg/lib -lavformat \
    -L$$PWD/ffmpeg/lib -lavcodec \
    -L$$PWD/ffmpeg/lib -lavutil \
    -L$$PWD/ffmpeg/lib -lswscale \
```

然后在需要用到ffmpeg的地方，需要引入c的头文件：

```C++
extern "C"
{
#include "libavcodec/avcodec.h"
#include "libavformat/avformat.h"
}
```

在运行的时候我们需要动态链接，即应该把ffmpeg的dll文件放到我们生成exe的目录下。同样可以通过qt来获得当前工作目录：

```C++
QString path;
path = QDir::currentPath();
qDebug()<<path;
```

我这里输出的结果如下：

```

```

## Log系统

```C++
#include <iostream>

extern "C"
{
#include "libavcodec/avcodec.h"
#include "libavformat/avformat.h"
#include "libavutil/log.h"
}

int main()
{
    av_log_set_level(AV_LOG_DEBUG);
    av_log(NULL, AV_LOG_ERROR, "hello, ffmepg");
    return 0;
}
```

FFmpeg的Log会涉及以下这些：

```
#include "libavutil/log.h"
av_log_set_level(AV_LOG_DEBUG);
av_log(NULL, AV_LOG_ERROR, "hello, ffmepg");
```

但是运行后发现日志并没有显示，我们查看一下`av_log()`的注释：

```c++
/**
 * Send the specified message to the log if the level is less than or equal
 * to the current av_log_level. By default, all logging messages are sent to
 * stderr. This behavior can be altered by setting a different logging callback
 * function.
 * @see av_log_set_callback
 *
 * @param avcl A pointer to an arbitrary struct of which the first field is a
 *        pointer to an AVClass struct or NULL if general log.
 * @param level The importance level of the message expressed using a @ref
 *        lavu_log_constants "Logging Constant".
 * @param fmt The format string (printf-compatible) that specifies how
 *        subsequent arguments are converted to output.
 */
```

默认的情况下，所有的日志被送到了stderr，如果设置一个回调函数则可以把日志回调到我们自己的函数里。

只要这样写就ok了：

```C++
#include <iostream>
extern "C"
{
#include "libavcodec/avcodec.h"
#include "libavformat/avformat.h"
#include "libavutil/log.h"
}

void logFF(void *, int level, const char *fmt, va_list ap)
{
    if (level <= AV_LOG_INFO)
        printf("%s \n", fmt);
}

int main()
{
    av_log_set_level(AV_LOG_DEBUG);
    av_log_set_callback(&logFF);
    av_log(NULL, AV_LOG_INFO, "hello, ffmepg");

    return 0;
}
```

## 文件的删除与重命名

和文件操作相关的api如下：

- FFmpeg 删除文件：avpriv_io_delete()
- FFmpeg 重命名文件：avpriv_io_move()
- FFmpeg 打开目录：avio_open_dir()
- FFmpeg 读取目录：avio_read_dir();
- FFmpeg 关闭目录：avio_close_dir()

**但是`avpriv_io_delete()`和`avpriv_io_move()`在windows上平台上是没有实现的。**

在其他平台上可以自行封装成：

```C++
void del_file(const char* path)
{
    // cannot use on windows
    int ret;
    ret = avpriv_io_delete(path);
    if (ret < 0)
    {
        printf("Error Code:%d \n", ret);
        av_log(NULL, AV_LOG_ERROR, "Failed to delete file %s \n", path);
    } else {
        av_log(NULL, AV_LOG_INFO, "Delete File Success！\n ");
    }
}

void move_file(char* src, char* dst)
{
    // cannot use on windows
    int ret;
    ret = avpriv_io_move(src, dst);
    if (ret < 0)
    {
        av_log(NULL, AV_LOG_ERROR, "Failed to Move File %s!\n ", src);
    } else {
        av_log(NULL, AV_LOG_INFO, "Success Move File %s!\n", src);
    }
}
```

使用FFmpeg文件操作API实现读取文件目录及输出文件目录List的相关代码如下：

```c++
// FFmpeg 目录操作
void ffmpegDir() {

    int ret;
    
    // 上下文
    AVIODirContext *dirCtx = NULL;
    AVIODirEntry *dirEntry = NULL;
    
    // 注意Windows下会返回-40，也就是Function not implement，方法未实现，也就是说windows下不支持此方法
    ret = avio_open_dir(&dirCtx, "include", NULL);  

    if (ret < 0) {
        // 输出错误日志
        printf("cant open dir，msg = %s", av_err2str(ret));
        return;
    }

    av_log(NULL, AV_LOG_INFO, "Open Dir Success!");

    while (1){
        ret = avio_read_dir(dirCtx, &dirEntry);
        if (ret < 0) {
            printf("cant read dir : %s", av_err2str(ret));
            // 防止内存泄漏
            goto __failed;
        }
        av_log(NULL, AV_LOG_INFO, "read dir success");
        if (!dirEntry) {
            break;
        }
        printf("Entry Name = %s", dirEntry->name);
        // 释放资源
        avio_free_directory_entry(&dirEntry);
    }
// 释放资源
__failed:
    avio_close_dir(&dirCtx);
}
```

## 多媒体文件基本概念

多媒体文件实际上是一个容器，内部有很多流(每种流是由不同的编码器编码的)。从流中读出的数据称为包，一个包里有多个帧。操作流数据的基本步骤如下：

```
解复用->获取流->读数据包->释放资源
```

几个重要的结构体

- AVFormatContext ：格式上下文，连接多个api的桥梁；
- AVStream：所有的音频流、视频流读取出来
- AVPacket：流中的所有数据包

### 打印音、视频信息

代码如下：

```C++
#include "stdafx.h"
extern "C"
{
#include "libavcodec/avcodec.h"
#include "libavformat/avformat.h"
#include "libavutil/log.h"
}

int main()
{
    const char* url = "test.mp4";
    int ret;
    AVFormatContext *fmt_ctx = NULL;
    ret = avformat_open_input(&fmt_ctx, url, NULL, NULL);
    if (ret < 0)
    {
        printf("Error code: %d: ", ret);
        printf("Cannot open file:%s \n", url);
        return -1;
    }
    av_dump_format(fmt_ctx, 0, url, 0);
    avformat_close_input(&fmt_ctx);
    return 0;
}
```

### 抽取音频数据

选取最佳的流：`av_find_best_stream()`和 将输出的包里的数据存入新文件：

```C++
void out_aac(const char* src, const char* dst)
{
    int ret;
    int audio_index; //确保是我们自己文件想要的流
    AVPacket pkt;
    AVFormatContext *fmt_ctx = NULL;
    ret = avformat_open_input(&fmt_ctx, src, NULL, NULL);
    if (ret < 0)
    {
        printf("Error code: %d: ", ret);
        av_log(NULL, AV_LOG_ERROR, "Cannot open file:%s \n", src);
        return;
    }
    FILE *dst_fd = fopen(dst, "wb");
    if (!dst_fd)
    {
        av_log(NULL, AV_LOG_ERROR, "Cannot open out file.\n");
        avformat_close_input(&fmt_ctx);
        return;
    }

    ret = av_find_best_stream(fmt_ctx, AVMEDIA_TYPE_AUDIO, -1, -1, NULL, 0);
    if (ret < 0)
    {
        av_log(NULL, AV_LOG_ERROR, "Cannot find the best stream!\n");
        avformat_close_input(&fmt_ctx);
        fclose(dst_fd);
        return;
    }

    audio_index = ret;

    av_init_packet(&pkt);
    while (av_read_frame(fmt_ctx, &pkt) >= 0)
    {
        if (pkt.stream_index == audio_index)
        {
            int len = fwrite(pkt.data, 1, pkt.size, dst_fd);
            if (len != pkt.size)
            {
                av_log(NULL, AV_LOG_WARNING, "Warning:length is not equal size!\n");
            }
        }
        av_packet_unref(&pkt);
    }
    av_log(NULL, AV_LOG_INFO, "Success get media info. \n");
    avformat_close_input(&fmt_ctx);
    if (dst_fd)
        fclose(dst_fd);
    av_log(NULL, AV_LOG_INFO, "Success output aac file. \n");
}
```

但是如果打开该文件就会发现：

![打不开aac文件](https://raw.githubusercontent.com/YunLambert/Blog_Pictures/master/img/20190903113800.png)

打不开aac文件，这是因为播放器在读取该文件的时候并不知道这个文件的信息，包括长度、如何解码等等的信息。所以一般的aac文件头部是会有这些信息的，我们也需要将其录入。

增加一部分代码：

```C++
#define ADTS_HEADER_LEN  7;

void adts_header(char *szAdtsHeader, int dataLen){

    int audio_object_type = 2;
    int sampling_frequency_index = 7;
    int channel_config = 2;

    int adtsLen = dataLen + 7;

    szAdtsHeader[0] = 0xff;         //syncword:0xfff                          高8bits
    szAdtsHeader[1] = 0xf0;         //syncword:0xfff                          低4bits
    szAdtsHeader[1] |= (0 << 3);    //MPEG Version:0 for MPEG-4,1 for MPEG-2  1bit
    szAdtsHeader[1] |= (0 << 1);    //Layer:0                                 2bits 
    szAdtsHeader[1] |= 1;           //protection absent:1                     1bit

    szAdtsHeader[2] = (audio_object_type - 1)<<6;            //profile:audio_object_type - 1                      2bits
    szAdtsHeader[2] |= (sampling_frequency_index & 0x0f)<<2; //sampling frequency index:sampling_frequency_index  4bits 
    szAdtsHeader[2] |= (0 << 1);                             //private bit:0                                      1bit
    szAdtsHeader[2] |= (channel_config & 0x04)>>2;           //channel configuration:channel_config               高1bit

    szAdtsHeader[3] = (channel_config & 0x03)<<6;     //channel configuration:channel_config      低2bits
    szAdtsHeader[3] |= (0 << 5);                      //original：0                               1bit
    szAdtsHeader[3] |= (0 << 4);                      //home：0                                   1bit
    szAdtsHeader[3] |= (0 << 3);                      //copyright id bit：0                       1bit  
    szAdtsHeader[3] |= (0 << 2);                      //copyright id start：0                     1bit
    szAdtsHeader[3] |= ((adtsLen & 0x1800) >> 11);           //frame length：value   高2bits

    szAdtsHeader[4] = (uint8_t)((adtsLen & 0x7f8) >> 3);     //frame length:value    中间8bits
    szAdtsHeader[5] = (uint8_t)((adtsLen & 0x7) << 5);       //frame length:value    低3bits
    szAdtsHeader[5] |= 0x1f;                                 //buffer fullness:0x7ff 高5bits
    szAdtsHeader[6] = 0xfc;
}
```

并在循环的每一AAC帧前加上这个ADTS头。
